/**
 * The method predecessor calculates the predecessor of a given number.
 *
 * @param i is the number the method returns the predecessor for
 * @return the predecessor of i
 */
int predecessor(int i) {
	return --i;
}

/**
 * The method square calculates the square of a given number.
 *
 * @param d is the number the method returns the square for
 * @return the square of d
 */
double square(double d) {
	return d * d;
}

/**
 * The method isEqual checks whether both given Strings are equal or not.
 *
 * @param first  is the first String
 * @param second is the second String
 * @return whether first and second are equal or not
 */
boolean isEqual(String first, String second) {
	return first.equals(second);
}

/**
 * The method echo writes a given String to the console. It is equal to
 * System.out.println(s).
 * 
 * @param s is the String which will be displayed in the console
 */
void echo(String s) {
	System.out.println(s);
	return;
}

/**
 * The method change calculates the optimal change for a given value of currency
 * in cents.
 *
 * @param cent is the given value of currency in cents
 */
void change(int cent) {
	String change = "";
	while (cent >= 200) {
		if (change == "") {
			change = "2\u20AC" + change; // \u20AC is the unicode representing the euro symbol
		} else {
			change = "2\u20AC, " + change;
		}
		cent -= 200;
	}
	if (cent >= 100) {
		if (change == "") {
			change = "1\u20AC" + change;
		} else {
			change = "1\u20AC, " + change;
		}
		cent -= 100;
	}
	if (cent >= 50) {
		if (change == "") {
			change = "50ct" + change;
		} else {
			change = "50ct, " + change;
		}
		cent -= 50;
	}
	while (cent >= 20) {
		if (change == "") {
			change = "20ct" + change;
		} else {
			change = "20ct, " + change;
		}
		cent -= 20;
	}
	if (cent >= 10) {
		if (change == "") {
			change = "10ct" + change;
		} else {
			change = "10ct, " + change;
		}
		cent -= 10;
	}
	if (cent >= 5) {
		if (change == "") {
			change = "5ct" + change;
		} else {
			change = "5ct, " + change;
		}
		cent -= 5;
	}
	while (cent >= 2) {
		if (change == "") {
			change = "2ct" + change;
		} else {
			change = "2ct, " + change;
		}
		cent -= 2;
	}
	if (cent == 1) {
		if (change == "") {
			change = "1ct" + change;
		} else {
			change = "1ct, " + change;
		}
	}
	System.out.println(change);
}

/**
 * The method changeShort calculates the optimal change for a given value of

 * @param cent is the given value of currency in cents
 */
void changeShort(int cent) {
	String change = "";
	int[] betraege = { 200, 100, 50, 20, 10, 5, 2, 1 };
	for (int b : betraege) {
		while (cent >= b) {
			cent = cent - b;
			if (b >= 100) {
				change = b / 100 + "\u20AC, " + change;
			} else {
				change = b + "ct, " + change;
			}
		}
	}
	System.out.println(change.substring(0, change.lastIndexOf(", ")));
}
