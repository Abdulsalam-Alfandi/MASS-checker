    /** Berechnet die Differenz zwischen zwei Daten in Stunden
     * @param a ist das erste Datum
     * @param b ist das zweite Datum
     * @return gibt die Differenz in Stunden zur端ck
     */
    public static int getTimeDifference(Calendar a, Calendar b) {
        long seconds = (a.getTimeInMillis() - b.getTimeInMillis());
        int hours = (int) (seconds / 3600);
        return hours;
    }

     * @param loan ist der Geldbetrag
     * @return gibt die Rate in Cents zur端ck
     */
    public static int emi(double interest, int payments, int loan) {
        int rate = (int) (loan*(interest*Math.pow(1+interest, payments))/Math.pow(1+interest, payments)-1);
        System.out.println(rate);
        return rate;
